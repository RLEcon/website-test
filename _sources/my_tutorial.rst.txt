import torch  # This is all you need to use both PyTorch and TorchScript!
print(torch.__version__)

class MyCell(torch.nn.Module):
    def __init__(self):
        super(MyCell, self).__init__()

    def forward(self, x, h):
        new_h = torch.tanh(x + h)
        return new_h, new_h
